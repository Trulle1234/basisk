In english / På engelska:

statements  : NEWLINE* expr (NEWLINE+ expr)* NEWLINE*

expr        : KEYWORD:låt IDENTIFIER EQ expr
            | comp-expr ((KEYWORD:och|KEYWORD:eller) comp-expr)*

comp-expr   : KEYWORD:inte comp-expr
            | arith-expr ((EE|NE|LT|GT|LTE|GTE) arith-expr)*

arith-expr  : term ((PLUS|MINUS) term)*

term        : factor ((MUL|DIV) factor)*

factor      : (PLUS|MINUS) factor
            | power

power       : call (POW factor)*

call        : atom (LPAREN (expr (COMMA expr)*)? RPAREN)?

atom        : INT|FLOAT|STRING|IDENTIFIER
            | LPAREN expr RPAREN
            | list-expr
            | if-expr
            | for-expr
            | while-expr
            | func-def

list-expr   : LSQUARE (expr (COMMA expr)*)? RSQUARE

if-expr     : KEYWORD:om expr KEYWORD:då expr
            | KEYWORD:om expr KEYWORD:då NEWLINE statements KEYWORD:sluta
            | KEYWORD:om expr KEYWORD:då expr (KEYWORD:annars_om expr KEYWORD:då expr)* (KEYWORD:annars expr)?
            | KEYWORD:om expr KEYWORD:då NEWLINE statements (KEYWORD:annars_om expr NEWLINE statements)* (KEYWORD:annars NEWLINE statements)? KEYWORD:sluta
